CREATE TABLE students ( student_id INT PRIMARY KEY, //Primary Key student_name VARCHAR(100) NOT NULL, exam_score INT CHECK (exam_score BETWEEN 0 AND 100), support BOOLEAN NOT NULL, // True / False date_of_birth DATE NOT NULL ); CREATE TABLE courses ( course_id CHAR(2) PRIMARY KEY //Primary Key (e.g., 'CS', 'MA') course_name VARCHAR(100) UNIQUE NOT ); CREATE TABLE exam_boards ( board_id VARCHAR(10) PRIMARY KEY, //Primary Key (e.g., 'AQA', 'OCR') board_name VARCHAR(50) UNIQUE NOT NULL ); CREATE TABLE teachers ( teacher_id CHAR(4) PRIMARY KEY, //Primary Key (e.g. TPAR’, TDAN’) teacher_name VARCHAR(100) UNIQUE NOT NULL ); CREATE TABLE course_offerings ( offering_id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY, //Surrogate PK course_id CHAR(2) NOT NULL REFERENCES courses(course_id) ON UPDATE CASCADE ON DELETE RESTRICT, board_id VARCHAR(10) NOT NULL REFERENCES exam_boards(board_id) ON UPDATE CASCADE ON DELETE RESTRICT, teacher_id CHAR(4) NOT NULL REFERENCES teachers(teacher_id) ON UPDATE CASCADE ON DELETE RESTRICT, CONSTRAINT uq_course_board UNIQUE (course_id, board_id) -- natural uniqueness ); CREATE TABLE enrollments ( student_id INT NOT NULL REFERENCES students(student_id) ON UPDATE CASCADE ON DELETE CASCADE, -- delete student → delete enrollments offering_id INT NOT NULL REFERENCES course_offerings(offering_id) ON UPDATE CASCADE ON DELETE CASCADE, -- delete offering → delete enrollments PRIMARY KEY (student_id, offering_id) -- composite PK
